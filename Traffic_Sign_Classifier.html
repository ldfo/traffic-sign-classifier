import cv2
def normalize(x):
    x = np.array(x)
    x_min, x_max = x.min(), x.max()
    return (x - x_min) / (x_max - x_min)

def preprocess(x):
#     to black and white
    x_bw = np.array([cv2.cvtColor(img, cv2.COLOR_RGB2GRAY) for img in x])
#     equalize and then normalize
    x_eq = np.array([cv2.equalizeHist(img) for img in x_bw])
    x_eq = x_eq[..., np.newaxis]
    return normalize(x_eq)

def network(x):
    w1 = tf.Variable(tf.truncated_normal([5, 5, 1, 28], mean=0, stddev=0.1, seed=666))
    b1 = tf.Variable(tf.zeros(28, dtype=tf.float32))
#   conv2d
    layer_1 = tf.nn.conv2d(x, w1, strides=[1, 1, 1, 1], padding='VALID')
#   Add bias
    layer_1 = tf.add(layer_1, b1)
#   Relu activation function
    layer_2 = tf.nn.relu(layer_1)
#   Max Pool layer
    layer_3 = tf.nn.max_pool(layer_2, ksize=[1, 1, 1, 1], strides=[1, 1, 1, 1], padding='SAME')
#   second conv2d
    w2 = tf.Variable(tf.truncated_normal([5, 5, 28, 16], mean=0, stddev=0.1, seed=666))
    b2 = tf.Variable(tf.zeros(16, dtype=tf.float32))
    layer_4 = tf.nn.conv2d(layer_3, w2, strides=[1, 1, 1, 1], padding='VALID')
    layer_4 = tf.add(layer_4, b2)
    layer_5 = tf.nn.relu(layer_4)
#   second max pool
    layer_6 = tf.nn.max_pool(layer_5, ksize=[1, 1, 1, 1], strides=[1, 1, 1, 1], padding='SAME')
# third conv2d
    w3 = tf.Variable(tf.truncated_normal([5, 5, 16, 10], mean=0, stddev=0.1, seed=666))
    b3 = tf.Variable(tf.zeros(10, dtype=tf.float32))
    layer_7 = tf.nn.conv2d(layer_6, w3, strides=[1, 2, 2, 1], padding='VALID')
    layer_8 = tf.add(layer_7, b3)
    layer_9 = tf.nn.relu(layer_8)
#   third max pool
    layer_10 = tf.nn.max_pool(layer_9, ksize=[1, 2, 2, 1], strides=[1, 1, 1, 1], padding='SAME')
    
#   flatten 
    flat = tf.contrib.layers.flatten(layer_10)
    fully = tf.contrib.layers.fully_connected(flat, num_outputs=256)
#     fully = tf.contrib.layers.dropout(fully, keep_prob=0.88)
    fully = tf.contrib.layers.fully_connected(fully, num_outputs=128)
#     fully = tf.contrib.layers.dropout(fully, keep_prob=0.88)
    fully = tf.contrib.layers.fully_connected(fully, num_outputs=64)
    output = tf.contrib.layers.fully_connected(fully, num_outputs=n_classes, activation_fn=None)
    print('input',x.shape)
    print('l1',layer_1.shape)
    print('l2',layer_2.shape)
    print('l3',layer_3.shape)
    print('l4',layer_4.shape)
    print('l5',layer_5.shape)
    print('l6',layer_6.shape)
    print('l7',layer_7.shape)
    print('l8',layer_8.shape)
    print('l9',layer_9.shape)
    print('l10',layer_10.shape)
    
    print('flat',flat.shape)
    
    
    return output

y = tf.placeholder(tf.int64, (None))

# Model
logits = network(x)
logits = tf.identity(logits, name='logits')

# Loss and Optimizer
calc_loss = tf.reduce_mean(tf.nn.sparse_softmax_cross_entropy_with_logits(logits=logits, labels=y))
optimizer = tf.train.AdamOptimizer()
training_operation = optimizer.minimize(calc_loss)

# Accuracy
inference_operation = tf.argmax(logits, 1)
get_probabilities = tf.nn.top_k(logits, k=5)
correct_pred = tf.equal(inference_operation, y)
calc_accuracy = tf.reduce_mean(tf.cast(correct_pred, tf.float32), name='accuracy')

saver = tf.train.Saver()
    
def evaluate_data(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    total_loss = 0
    inference_data = np.array([])
    sess = tf.get_default_session()
    for offset in range(0, num_examples, batch_size):
        end = offset + batch_size
        batch_x, batch_y = X_data[offset:end], y_data[offset:end]
        acc, loss, inference = sess.run([calc_accuracy, calc_loss, inference_operation], feed_dict={x: batch_x, y: batch_y})
        total_accuracy += (acc * len(batch_x))
        total_loss += (loss * len(batch_x))
        inference_data = np.append(inference_data, inference)
    return total_accuracy / num_examples, total_loss / num_examples, inference_data

def print_progress(epoch, X_tr, y_tr, X_val, y_val, start_time):

    cols = [float(epoch), tr_acc, val_acc, tr_loss, val_loss, time.time() - start_time]

Step 4 (Optional): Visualize the Neural Network's State with Test Images¶

This Section is not required to complete but acts as an additional excersise for understaning the output of a neural network's weights. While neural networks can be a great learning device they are often referred to as a black box. We can understand what the weights of a neural network look like better by plotting their feature maps. After successfully training your neural network you can see what it's feature maps look like by plotting the output of the network's weight layers in response to a test stimuli image. From these plotted feature maps, it's possible to see what characteristics of an image the network finds interesting. For a sign, maybe the inner network feature maps react with high activation to the sign's boundary outline or to the contrast in the sign's painted symbol.

Provided for you below is the function code that allows you to get the visualization output of any tensorflow weight layer you want. The inputs to the function should be a stimuli image, one used during training or a new one you provided, and then the tensorflow variable name that represents the layer's state during the training process, for instance if you wanted to see what the LeNet lab's feature maps looked like for it's second convolutional layer you could enter conv2 as the tf_activation variable.

For an example of what feature map outputs look like, check out NVIDIA's results in their paper End-to-End Deep Learning for Self-Driving Cars in the section Visualization of internal CNN State. NVIDIA was able to show that their network's inner weights had high activations to road boundary lines by comparing feature maps from an image with a clear path to one without.
